1. Architecture Overview
1.1 Clean Architecture with BLoC
NOVA's Flutter codebase follows Clean Architecture principles, separating the application into three
concentric layers: Presentation, Domain, and Data. State management is handled by the BLoC
(Business Logic Component) pattern, which keeps UI widgets reactive and testable.
•​
•​
•​

Presentation Layer: Flutter widgets, BLoC classes, and UI state. Widgets observe BLoC states
and dispatch events. No business logic lives here.
Domain Layer: Use cases (interactors), entity classes, and repository interfaces. Pure Dart,
zero Flutter dependency. This is the application's business logic.
Data Layer: Repository implementations, data sources (remote API, local SQLite, TFLite
models, camera stream), and data transfer objects (DTOs).

The BLoC pattern replaces traditional callbacks with streams, making the asynchronous nature of
camera processing, TTS, and network calls predictable and testable. Each feature module (obstacle
detection, OCR, scene description, currency, face recognition) has its own BLoC.

double _estimateDistance(String label, double boxWidthNorm, int frameWidth) {
final refWidth = _referenceWidths[label] ?? 0.5;
// Pinhole camera model: distance ~ (focalLength * refWidth) / pixelWidth
// focalLength estimated at 600px for typical 8MP phone camera
const double focalLength = 600.0;
final pixelWidth = boxWidthNorm * frameWidth;
if (pixelWidth <= 0) return 999.0;
return (focalLength * refWidth) / pixelWidth;
}

if (lastAlert != null && now.difference(lastAlert) < _suppressionPeriod) {

await _tts.speak(

priority: urgent ? TtsPriority.critical : TtsPriority.high,
interrupt: urgent,

language: language,
success: true,
blockCount: recognised.blocks.length,

String _detectLanguage(List<TextBlock> blocks) {
// Simple heuristic: presence of French diacritics
final frenchChars = RegExp(r'[àâäéèêëîïôùûüÿçœæ]', caseSensitive: false);
int frenchScore = 0;
for (final block in blocks) {
if (frenchChars.hasMatch(block.text)) frenchScore++;
}
return frenchScore > blocks.length * 0.3 ? 'fr-CM' : 'en-CM';
}
Future<void> dispose() => _recognizer.close();
}

10. Offline Sync Engine
10.1 Overview
When the device has no network, NOVA writes all usage events and feedback to the local SQLite
database with synced = false. A background SyncService watches for connectivity restoration and
automatically flushes the local queue to the backend. The sync is idempotent; each event has a UUID
generated on the device, so if the network drops mid-batch, re-sending the same batch causes no
duplicate inserts on the server.

@singleton class SyncService {
final ConnectivityService _connectivity;
final AppDatabase _db;
final DioClient _dio;
StreamSubscription? _connectivitySubscription;
bool _isSyncing = false;
static const int _maxBatchSize = 50;
static const int _maxRetries = 3;
static const Duration _retryBackoffBase = Duration(seconds: 2);
SyncService(this._connectivity, this._db, this._dio);
/// Call this once from main.dart after dependency injection is ready.
void startWatching() {
_connectivitySubscription = _connectivity.onConnectivityChanged
.where((isConnected) => isConnected) // only fire on reconnection
.listen((_) => _triggerSync());
}
/// Also call this manually on app foreground resume.
Future<void> syncNow() => _triggerSync();
Future<void> _triggerSync() async {
if (_isSyncing) return; // prevent concurrent sync runs
_isSyncing = true;
try {
await _syncUsageEvents();
await _syncFeedback();
await _drainPendingQueue();
} finally {
_isSyncing = false;
}
}
Future<void> _syncUsageEvents() async {
final unsynced = await _db.getUnsyncedEvents();
if (unsynced.isEmpty) return;
for (int i = 0; i < unsynced.length; i += _maxBatchSize) {
final batch = unsynced.skip(i).take(_maxBatchSize).toList();
final success = await _postWithRetry(
endpoint: '/logs/sync',
payload: {
'events': batch.map((e) => {
'id': e.serverId,
'module_id': e.moduleId,
'timestamp': e.timestamp.toIso8601String(),
'outcome': e.outcome,
'confidence_score': e.confidenceScore,
}).toList(),
},
);
if (success) { await _db.markEventsSynced(batch.map((e) => e.id).toList());
}
}
}

Future<void> _syncFeedback() async {
final unsynced = await _db.getUnsyncedFeedback();
if (unsynced.isEmpty) return;
for (int i = 0; i < unsynced.length; i += _maxBatchSize) {
final batch = unsynced.skip(i).take(_maxBatchSize).toList();

'feedbacks': batch.map((f) => {

Future<void> _drainPendingQueue() async {
final pending = await _db.getPendingQueueItems();
for (final item in pending) {
if (item.retryCount >= _maxRetries) { await
_db.deletePendingQueueItem(item.id); continue; }
final success = await _postWithRetry(endpoint: item.endpoint, payload:
item.payloadJson, maxAttempts: 1);
if (success) { await _db.deletePendingQueueItem(item.id); } else { await
_db.incrementRetryCount(item.id); }
}
}
Future<bool> _postWithRetry({required String endpoint, required dynamic
payload, int maxAttempts = _maxRetries}) async {
for (int attempt = 0; attempt < maxAttempts; attempt++) {
try { await _dio.client.post(endpoint, data: payload); return true; }
catch (e) { if (attempt < maxAttempts - 1) await
Future.delayed(_retryBackoffBase * (1 << attempt)); }
} return false;
}
void dispose() { _connectivitySubscription?.cancel(); }
}
